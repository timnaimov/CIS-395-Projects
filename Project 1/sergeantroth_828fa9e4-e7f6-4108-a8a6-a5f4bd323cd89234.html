use hoteldb;

-- create table HOTEL with below attributes, respective data types, primary key

CREATE TABLE HOTEL(

hotelNo numeric primary key,

name varchar(40),

address varchar(40),

city varchar(200)

);

-- create table ROOM with below attributes, respective data types, primary and foreign key

CREATE TABLE ROOM(

roomNo numeric Primary Key,

hotelNo numeric References HOTEL,

type varchar(20),

price dec(9,2)

);

-- create table GUEST with below attributes, respective data types, primary key

CREATE TABLE GUEST(

guestNo numeric primary key,

name varchar(40),

address varchar(40)

);

-- create table BOOKING with below attributes, respective data types, foreign key

CREATE TABLE BOOKING(

hotelNo numeric REFERENCES Hotel,

guestNo numeric REFERENCES Guest,

dateFrom date NOT NULL,

dateTo date,

roomNo numeric REFERENCES ROOM

);

-- select columns h.name, h.address, r.roomNo, r.type, r.price

select h.name, h.address, r.roomNo, r.type, r.price

from room as r -- from room table with an alias name r

inner join hotel as h -- inner join hotel to room with an alias name h

on r.hotelNo = h.hotelNo -- on condition to join both the tables

where type = 'Family'; -- where condition to get the Family room

Question 2)

-- select columns h.name, round(avg(r.price),2) as 'average price'

select h.name, round(avg(r.price),2) as 'average price'

from room as r -- from room table with an alias name r

inner join hotel as h -- inner join hotel to room with an alias name h

on r.hotelNo = h.hotelNo -- on condition to join both the tables

group by h.name; -- group by column hotel name
